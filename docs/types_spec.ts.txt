// Rivalyze frontend types — owner Dhwani; mirrors backend models.py 1:1 (that file wins conflicts)
export type Status = 'queued' | 'running' | 'completed' | 'failed';
export interface RunEvent { t: number; agent: string; msg: string; }
export interface RunStatus {
  job_id: string; status: Status; current_stage: string;
  events: RunEvent[]; lane_stats: Record<string, number>;
  run_id: string | null; error: string | null;
}
export interface AnalyzeResponse { job_id: string; status: 'queued' | 'completed'; }
export interface EvidenceSource {
  id: string; source_type: 'news' | 'pricing' | 'review' | 'web' | 'document';
  source_name: string; url: string; snippet: string; source_date: string; agent: string;
}
export interface EvidenceResponse { claim_ref: string; sources: EvidenceSource[]; }
export interface HistoryRow {
  job_id: string; company: string;
  threat_level: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  confidence: number; created_at: string;
}
export interface Opportunity { text: string; evidence_ids: string[]; claim_ref: string; }
export interface Recommendation {
  action: string; rationale: string; confidence: number;
  evidence_ids: string[]; claim_ref: string;
}
export interface Swot { strengths: string[]; weaknesses: string[]; opportunities: string[]; threats: string[]; }
export interface SentimentScore { score: number; label: 'POSITIVE' | 'NEUTRAL' | 'NEGATIVE'; }
export interface H2HCell { value: string; claim_ref?: string; source_date?: string; }
export interface H2HRow { metric: string; you: string; rivals: Record<string, H2HCell>; }
export interface CompetitiveReport {
  company: string; threat_level: HistoryRow['threat_level'];
  executive_summary: string; swot: Swot;
  sentiment: Record<string, SentimentScore>; head_to_head: H2HRow[];
  opportunities: Opportunity[]; recommendations: Recommendation[];
  low_signal_findings: string[]; analysis_date: string;
}
