import type { AnalyzeResponse, RunStatus, CompetitiveReport, EvidenceResponse, HistoryRow } from './types'

const BASE = import.meta.env.VITE_API_URL ?? 'http://localhost:8000'
const HEADERS = { 'Content-Type': 'application/json', Authorization: `Bearer ${import.meta.env.VITE_TOKEN ?? ''}` }

async function req<T>(path: string, init?: RequestInit): Promise<T> {
  const r = await fetch(`${BASE}${path}`, { headers: HEADERS, ...init })
  if (!r.ok) throw new Error(`${r.status} ${path}`)
  return r.json() as Promise<T>
}
export const analyze = (body: { company: string; domain?: string; idea?: string }) =>
  req<AnalyzeResponse>('/api/v1/analyze', { method: 'POST', body: JSON.stringify(body) })
export const getRun = (jobId: string) => req<RunStatus>(`/api/v1/runs/${jobId}`)
export const getReport = (runId: string) => req<CompetitiveReport>(`/api/v1/reports/${runId}`)
export const getEvidence = (claimRef: string, runId: string) =>
  req<EvidenceResponse>(`/api/v1/evidence/${encodeURIComponent(claimRef)}?run_id=${runId}`)
export const getHistory = (company?: string) =>
  req<HistoryRow[]>(`/api/v1/history${company ? `?company=${encodeURIComponent(company)}` : ''}`)
